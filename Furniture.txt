{
    public class Furniture
    {
        public string Name { get; set; }
        public string Manufacturer { get; set; }
        public string Cost { get; set; }

        public Furniture(string name, string manufacturer, string cost)
        {
            Name = name;
            Manufacturer = manufacturer;
            Cost = cost;
        }
    }

    protected void Page_Load(object sender, EventArgs e)
    {
        if (!IsPostBack)
        {
            // Create three session objects and store Furniture objects in them
            Furniture furniture1 = new Furniture("Chair 1", "XYZ", "1000 Rupees");
            Furniture furniture2 = new Furniture("Chair 2", "ABC", "1200 Rupees");
            Furniture furniture3 = new Furniture("Chair 3", "PQR", "1500 Rupees");

            Session["Furniture1"] = furniture1;
            Session["Furniture2"] = furniture2;
            Session["Furniture3"] = furniture3;

            // Add the names of the Furniture objects to the ListBox
            ListBox1.Items.Add(furniture1.Name);
            ListBox1.Items.Add(furniture2.Name);
            ListBox1.Items.Add(furniture3.Name);
        }
    }

    protected void Button1_Click(object sender, EventArgs e)
    {
        // Get the selected Furniture object from the ListBox
        int selectedIndex = ListBox1.SelectedIndex;
        Furniture selectedFurniture = null;

        if (selectedIndex == 0)
        {
            selectedFurniture = (Furniture)Session["Furniture1"];
        }
        else if (selectedIndex == 1)
        {
            selectedFurniture = (Furniture)Session["Furniture2"];
        }
        else if (selectedIndex == 2)
        {
            selectedFurniture = (Furniture)Session["Furniture3"];
        }

        // Display the information of the selected Furniture object in the Label
        if (selectedFurniture != null)
        {
            Label1.Text = "Name: " + selectedFurniture.Name + "<br/>" +
                          "Manufacturer: " + selectedFurniture.Manufacturer + "<br/>" +
                          "Cost: " + selectedFurniture.Cost;
        }
        else
        {
            Label1.Text = "No furniture selected.";
        }
    }
}
